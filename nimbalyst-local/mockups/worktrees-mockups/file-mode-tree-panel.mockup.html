import { OAuth2Client } from 'google-auth-library';
import { Octokit } from '@octokit/rest';

export interface OAuthConfig {
  clientId: string;
  clientSecret: string;
  redirectUri: string;
}

export class OAuthService {
  private googleClient: OAuth2Client;
  private githubClient: Octokit;

  constructor(
    private googleConfig: OAuthConfig,
    private githubConfig: OAuthConfig
  ) {
    this.googleClient = new OAuth2Client(
      googleConfig.clientId,
      googleConfig.clientSecret,
      googleConfig.redirectUri
    );
  }

  async getGoogleAuthUrl(): Promise<string> {
    return this.googleClient.generateAuthUrl({
      access_type: 'offline',
      scope: ['profile', 'email']
    });
  }

  async getGithubAuthUrl(): Promise<string> {
    const params = new URLSearchParams({
      client_id: this.githubConfig.clientId,
      redirect_uri: this.githubConfig.redirectUri,
      scope: 'user:email'
    });
    return `https://github.com/login/oauth/authorize?${params}`;
  }

  async handleGoogleCallback(code: string) {
    const { tokens } = await this.googleClient.getToken(code);
    this.googleClient.setCredentials(tokens);
    // Verify and return user info
  }
}
